### React Typescript Examples:

```.tsx
import * as React from 'react';
import { render } from 'react-dom';
  
interface Todo {
  userId: number;
  id: number;
  title: string;
  completed: boolean;
}
async function fetchTodos(request: RequestInfo): Promise<Todo[]> {
  const response = await fetch(request);
  const body = await response.json();
  return body;
}
  
enum Color {
  blue = 'Blue Color', // value = 'Label'
  orange = 'Orange Color'
}
type ColorKey = keyof typeof Color;
  
interface BaseProps {
  name: string
}
interface AppProps extends BaseProps {
  pageId?: string
}
export default function App({ name }: AppProps) {
  const [color, setColor] = React.useState(Object.keys(Color)[0]);
  const [todos, setTodos] = React.useState<Todo[]>([]);
  React.useEffect(() => {
    (async () => {
      const arr: Todo[] = await fetchTodos('https://jsonplaceholder.typicode.com/todos')
      setTodos(arr);
    })();
  }, [])
  return (
    <div>
      <h3 style={{ color }}>Typescript, Hello {name ?? 'DefaultValue'}</h3>
      <select onChange={(ev: React.ChangeEvent<HTMLSelectElement>) => setColor(ev.target.value)}>
        {Object.keys(Color).map((color) => (
          <option key={color} value={color}>
            {Color[color as ColorKey]}
          </option>
        ))}
      </select>
      <h3>Fetched todo items:</h3>
      {todos.map((item) => <div key={item.id}>{item.id} - {item.title}</div>)}
    </div>
  );
}
render(<App name="App" />, document.getElementById("root"));
```
