## Stencil 1.14

### Stencil Component

```.js
import { Component, Prop, State, h } from "@stencil/core";
  
export interface Item {
  title: string;
  done?: boolean;
}
@Component({
  tag: "my-component",
  styleUrl: "my-component.scss",
  shadow: true
})
export class MyComponent {
  @Prop() initialItems: Item[];
  @State() items: Item[] = [{ title: "Item 1" }];
  
  fetchMore = () => {
    this.items = [...this.items, { title: "Fetched item" }];
  };
  
  render() {
    return (
      <div>
        {this.items.length > 0 && (
          <ul>
            {this.items.map(item => (
              <li>{item.title}</li>
            ))}
          </ul>
        )}
        <button onClick={this.fetchMore}>Fetch More</button>
      </div>
    );
  }
}
```
