### Component with Composition API (34 LOC)

```.js
<template>
  <v-fragment>
    <ul v-if="items.length > 0">
      <li v-for="(item, index) in items" :key="index">
        {{ item.title }}
        <a v-on:click="deleteItem(index)">X</a>
      </li>
    </ul>
    <input v-model="title" placeholder="Task Title">
    <button v-on:click="addItem">Add Item</button>
  </v-fragment>
</template>
  
<script>
import { ref, computed, onMounted } from 'vue';
  
export default {
  setup(props) {
    onMounted(() => console.log('component mounted!'));
    const items = ref([{ title: 'Item 1' }]);
    const title = ref('');
    const addItem = () => {
      items.value = [...items.value, { title: title.value }];
      title.value = '';
    };
    const deleteItem = (index) => items.value = items.value.filter((_, idx) => idx !== index);
    return {
      items,
      title,
      addItem,
      deleteItem
    };
  }
};
</script>
```
