## React 16 in Typescript

### Functional Component (33 LOC)

```.tsx
import * as React from 'react';
  
export interface Item {
  id: string;
  title: string;
  done?: boolean;
}
export interface Props {
  initialItems?: Item[];
}
export default function TodoList({ initialItems = [] }: Props): React.ReactElement {
  const [items, setItems] = React.useState<Item[]>(initialItems);
  const [title, setTitle] = React.useState('');
  
  const addItem = () => {
    setItems([...items, { id: `${Math.random()}`, title }]);
    setTitle('');
  };
  const deleteItem = (index: number) => setItems(items.filter((_, idx) => idx !== index));
  return (
    <>
      {items.length > 0 && (
        <ul>
          {items.map((item, index) => (
            <li key={item.id}>
              {item.title} <button onClick={() => deleteItem(index)}>X</button>
            </li>
          ))}
        </ul>
      )}
      <input value={title} placeholder="Task Title" onChange={(e) => setTitle(e.target.value)} />
      <button onClick={addItem}>Add Item</button>
    </>
  );
}
```
