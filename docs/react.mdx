## React 16 in Typescript

### Functional Component

```.tsx
import * as React from 'react'

export interface Item {
  title: string
  done?: boolean
}
export interface Props {
  initialItems?: Item[]
}
  
export default function TodoList({ initialItems = [] }: Props): React.ReactElement {
  const [items, setItems] = React.useState<Item[]>(initialItems)

  const fetchMore = () => {
    setItems([...items, { title: 'Fetched item' }])
  }
  return (
    <>
      {items.length > 0 && (
        <ul>
          {items.map((item) => (
            <li>{item.title}</li>
          ))}
        </ul>
      )}
      <button onClick={fetchMore}>Fetch More</button>
    </>
  )
}
```

### Class

```.tsx
interface Props {
  value: number
}
interface State {
  text: string
}
  
class MyComponent extends React.Component<Props, State> {
  static defaultProps: Props = {
    value: 0
  }
  state: State = { text: 'Example' }

  render() {
    return <div>Syntax {this.props.value} {this.state.text}</div>
  }
}
```
