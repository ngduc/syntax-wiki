### Component with Typescript (40 LOC)

```.js
import { Component, OnInit, OnDestroy, Input } from '@angular/core';
  
interface Item {
  title: string;
  done?: boolean;
}
@Component({
  selector: 'app-root',
  template: `
    <ul *ngIf="items.length > 0">
      <li *ngFor="let item of items; index as idx; trackBy: trackByFn">
        {{ item.title }}
        <a (click)="deleteItem(idx)">X</a>
      </li>
    </ul>
    <input value="{{ title }}" #titleInput (keyup)="onKey(titleInput.value)" />
    <button (click)="addItem()">Fetch More</button>
  `,
  // templateUrl: './app.component.html', // for template file
  styleUrls: ['./app.component.css']
})
export class AppComponent implements OnInit, OnDestroy {
  @Input(100) maxItems: number;
  ngOnInit(): void {}
  ngOnDestroy(): void {}
  
  title = '';
  items: Item[] = [{ title: 'Item 1' }];
  trackByFn(index: number, item: Item) {
    return index; // or: item.id
  }
  
  onKey(value: string) {
    this.title = value;
  }
  addItem(): void {
    this.items = [...this.items, { title: this.title }];
    this.title = '';
  }
  deleteItem(idx: number): void {
    this.items = this.items.filter((_, itemIdx) => itemIdx !== idx);
  }
}
```
