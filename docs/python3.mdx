```.js
import json
import math
from dataclasses import dataclass, asdict
  
@dataclass
class Life:
  message: str = ''
  seed: float = round(math.pi, 2)  # 3.14
  answer: int = 0
  
  def getAnswerOfLife(self) -> int:
    arr = f'{self.seed}'.split('.')          # '3.14' -> ['3', '14']
    self.answer = int(arr[0]) * int(arr[1])  # 3 * 14 = 42
    return self.answer
  
  def shout(self) -> str:
    charArr = list(self.message)  # string to array of chars
    msg = ''
    for idx, item in enumerate(charArr):
      upperChar = chr(ord(charArr[idx]) - 32)
      msg = msg + (' ' if charArr[idx] == ' ' else upperChar)
    return f'{msg} : {asdict(self)}'
  
myLife = Life(message = 'hello world')
myLife.getAnswerOfLife()
print(myLife.shout())
```
