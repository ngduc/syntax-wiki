### Syntax

```.js
import json
import math
from dataclasses import dataclass, asdict
  
@dataclass
class Life:
  message: str = ''
  seed: float = round(math.pi, 2)  # 3.14
  answer: int = 0
  
  def getAnswerOfLife(self) -> int:
    arr = f'{self.seed}'.split('.')          # '3.14' -> ['3', '14']
    self.answer = int(arr[0]) * int(arr[1])  # 3 * 14 = 42
    return self.answer
  
  def shout(self) -> str:  # shout in UPPERCASE (not using Python .upper)
    msg = ''
    for char in self.message:
      upperChar = chr(ord(char) - 32)
      msg = msg + (' ' if char == ' ' else upperChar)
    return f'{msg} : {asdict(self)}'
  
myLife = Life(message = 'hello world')
myLife.getAnswerOfLife()
print(myLife.shout())
```
